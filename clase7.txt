/*Ejercicio 1*/                        

SELECT title, rating 
  FROM film 
 WHERE length <= ALL (SELECT length 
                        FROM film); 
 

/*Ejercicio 2*/                        
SELECT title
  FROM film f1 
 WHERE length < ALL (SELECT length 
                       FROM film f2
                      WHERE f2.film_id <> f1.film_id);
                      

              
/*Ejercicio 3*/

SELECT customer_id, first_name, last_name, a.address
	(SELECT MIN(amount) FROM payment p
		WHERE p.customer_id = c.customer_id ) as 'amount'
FROM customer c, address a 
WHERE c.address_id = a.address_id ;
	

SELECT first_name, last_name, a.address,
       (SELECT DISTINCT amount 
          FROM payment 
         WHERE c.customer_id = payment.customer_id 
           AND amount <= ALL (SELECT amount 
                                FROM payment 
                               WHERE c.customer_id = payment.customer_id))
	   AS min_amount 
 FROM customer c, address a 
 WHERE  c.address_id = a.address_id;
